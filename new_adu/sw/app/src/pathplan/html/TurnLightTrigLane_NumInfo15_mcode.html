1function Lane_NumInfo_Out  = TurnLightTrigLane_NumInfo(TrigLaneChgEn,TurnLight,TurnLightAdjustEn,ResetEdg,Lane_NumInfo,Veh_State,Navi_InfoTurn,TurnFobDis,DisEngFlg)
 2%TurnLight Left 1 Right 2
 3persistent TargetLaneNum LeftTurnRisEdg RightTurnRisEdg
 4if isempty(TargetLaneNum)||(Lane_NumInfo(1)<0.5)||(Lane_NumInfo(3)<0.5)||(ResetEdg>0.5)
 5    TargetLaneNum = 2;
 6    LeftTurnRisEdg = 0;
 7    RightTurnRisEdg = 0;
 8end
 9    Lane_NumInfo_Out = Lane_NumInfo; 
10    if abs(TrigLaneChgEn) > 0.5
11        if TrigLaneChgEn > 0
12            Lane_NumInfo_Out(3) = Lane_NumInfo(3) + TrigLaneChgEn;
13            if Lane_NumInfo_Out(3)>Lane_NumInfo(1)
14                Lane_NumInfo_Out(3) = Lane_NumInfo(1);
15            end                
16        else
17            Lane_NumInfo_Out(3) = Lane_NumInfo(3) + TrigLaneChgEn;
18            if Lane_NumInfo_Out(3) < 1
19                Lane_NumInfo_Out(3) = 1;
20            end  
21        end      
22    else           
23    end
24    
25    if TurnLightAdjustEn > 0.5 && (Lane_NumInfo(1)>0.5) && (Lane_NumInfo(3)>0.5)
26        TunEnbResTemp = 1;
27        if Navi_InfoTurn(1,2) < 0.5 || Navi_InfoTurn(2,2) < 0.5
28            % TunTrig will be forbidden
29            TunEnbResTemp = 0;  
30        else
31            TunOffset = Navi_InfoTurn(2,1) - Navi_InfoTurn(1,1);
32            if TunOffset <= 0 || TunOffset > 100
33                TunEnbResTemp = 0;
34            end                
35        end
36        InTurnFlg = 0;
37        if TunEnbResTemp > 0.5
38            if ((Veh_State(4) - Navi_InfoTurn(1,1) + TurnFobDis) > 0) && ((Veh_State(4) - Navi_InfoTurn(2,1)) < 0)
39                InTurnFlg = 1;
40            end
41        end
42        if InTurnFlg < 0.5 && DisEngFlg < 0.5
43            if abs(TurnLight-1)<1e-1 
44                if LeftTurnRisEdg < 0.5
45                    TargetLaneNum = Lane_NumInfo(2) - 1;
46                    LeftTurnRisEdg = 1;
47                    if TargetLaneNum < 1
48                        TargetLaneNum = 1;
49                    end         
50                end
51            elseif  abs(TurnLight-2)<1e-1 
52                if RightTurnRisEdg < 0.5
53                    TargetLaneNum = Lane_NumInfo(2) + 1;
54                    RightTurnRisEdg = 1;
55                    if TargetLaneNum > Lane_NumInfo(1)
56                        TargetLaneNum = Lane_NumInfo(1);
57                    end  
58                end
59            else           
60                LeftTurnRisEdg = 0;
61                RightTurnRisEdg = 0;
62            end
63            Lane_NumInfo_Out(3) = TargetLaneNum;
64        else
65            TargetLaneNum = Lane_NumInfo(3);
66            Lane_NumInfo_Out(3) = TargetLaneNum;
67        end
68    end
69    
70end
