464function [Dis,Dir] = DisCal2Lane(LaneAng,x1,y1,xobj,yobj)
465%      A*x + B*y + C
466%d = -----------------
467%      sqrt(A^2 + B^2)
468%
469%Dir Front 1 Rear 0 Left 1 Right 0
470%example 11 FL 10 FR 1 RL 0 RR
471%Xdirection Front and Rear Ydirection Left and Right
472    
473    while(1)
474       if LaneAng < -pi
475           LaneAng = LaneAng + 2*pi;
476       elseif LaneAng > pi
477           LaneAng = LaneAng - 2*pi;
478       else           
479            if LaneAng > pi/2
480                LaneAng = LaneAng - pi;
481            elseif LaneAng < -pi/2
482                LaneAng = LaneAng + pi;
483            else
484            end           
485            break;
486       end
487    end
488%Convert LaneAng to -pi/2 - pi/2    
489
490    if abs(sin(LaneAng))>1.414
491        InvK = cot(LaneAng);
492        A = 1;B = -InvK;C = InvK*y1 - x1;
493    else
494        K = tan(LaneAng);
495        A = K;B = -1;C = y1-K*x1;
496    end
497    
498    Dis = (A*xobj + B*yobj + C)/sqrt(A^2 + B^2);
499    %Left or Right
500    if abs(A)<abs(B)
501        if (B * Dis > 0)
502            Dir = 0;
503            Dis = abs(Dis);
504        else
505            Dir = 1;
506            Dis = -1*abs(Dis);
507        end    
508    else
509        if (A * Dis > 0)
510            %Right
511            Dir = 0;
512            Dis = -1*abs(Dis);

524function [Dis,Dir] = DisCal2Arc(kur,xobj,yobj)
525    if abs(kur) < 1e-3
526        % Treat as a straight lane
527        [Dis,Dir] = DisCal2Lane(0,0,0,xobj,yobj);
528    else
529        % Treat as a circle
530        CircleX = 0;CircleY = 1/kur;
531        Dis = sqrt((xobj-CircleX)^2+(yobj-CircleY)^2)-abs(CircleY);
532        if kur*Dis>0
533            %Right
534            Dir = 1;
535            Dis = -1*abs(Dis);
536        else
537            %Left
538            Dir = 0;
539            Dis = abs(Dis);
540        end        
541    end
542end
