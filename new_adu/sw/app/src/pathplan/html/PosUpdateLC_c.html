  516       }
  517     }
  518   
  519     if ((b_PointCalEn < 0.1) && (PathPlanRes > 0.1)) {
  520       b_PointCalEn = 1.0;
  521       if (b_PrePointCalEn < 0.1) {
  522         b_PrePointCalEn = 1.0;
  523         FirstTrigFlg = 1;
  524       }
  525     }
  526   
  527     if (fabs(PathPlanRes - PPathPlanRes) > 0.1) {
  528       PPathPlanRes = PathPlanRes;
  529     }
  530   
  531     *xpositionNex = 0.0;
  532     *ypositionNex = 0.0;
  533   
  534     /*  Not supposed to work */
  535     *PreTimeAcu = 2.0;
  536     if (b_PointCalEn > 0.1) {
  537       if (FirstTrigFlg > 0.1) {
  538         b_qiConfigCal(c_x_in, c_y_in, c_angle_in, xoptres[0], c_y_out, c_angle_out,
  539                       &Px_qitemp, &Py_qitemp, &Ptheta_qitemp);
  540         c_ElementaryPathPlot(c_x_in, c_y_in, c_angle_in, Px_qitemp, Py_qitemp,
  541                              Ptheta_qitemp, &b_DeltaTemp[0], &PreDis,
  542                              &b_ArclengthTol[0]);
  543         c_ElementaryPathPlot(Px_qitemp, Py_qitemp, Ptheta_qitemp, xoptres[0],
  544                              c_y_out, c_angle_out, &b_DeltaTemp[1], &PreDis,
  545                              &b_ArclengthTol[1]);
  546         PToLeftOrRight = ToLeftOrRight;
  547   
  548         /*              Pxoptres = xoptres; */
  549         for (i7 = 0; i7 < 2; i7++) {
  550           PDeltaTemp[i7] = b_DeltaTemp[i7];
  551           PArclengthTol[i7] = b_ArclengthTol[i7];
  552         }
  553   
  554         b_Pangle_in = c_angle_in;
  555         b_Px_in = c_x_in;
  556         b_Py_in = c_y_in;
  557         b_Pangle_out = c_angle_out;
  558         b_Px_out = xoptres[0];
  559         b_Py_out = c_y_out;
  560         b_PTime = 0.0;
  561         c_PT_long = xoptres[1] * 2.0 * (b_ArclengthTol[0] + b_ArclengthTol[1]) /
  562           (xoptres[2] + Info_Me[2]);
  563         d_PV_tar = xoptres[2];
  564         b_PV_me = Info_Me[2];
  565         PPreTmTransCnt = 50.0;
  566         PInLaneLCTrans = 0.0;
  567         PxpositionNex = 0.0;
  568         PypositionNex = 0.0;
  569       } else {
  570         b_PTime += 0.01;
  571       }
  572   
  573       /*  Transition of Preview Time */
  574       if (PPreTmTransCnt > 0.5) {
  575         PPreTmTransCnt--;
  576         PreTime = (50.0 - PPreTmTransCnt) / 50.0 + 1.0;
  577       }
  578   
  579       FirstTrigFlg = 0;
  580       Kur = 0.0;
  581       V_tarPos = 0.0;
  582       PreDis = 0.0;
  583       while ((FirstTrigFlg < 0.5) || ((PreDis < 1.0) && (FirstTrigFlg < 10))) {
  584         FirstTrigFlg++;
  585         PreDis = b_PTime + PreTime * (double)FirstTrigFlg;
  586         *PreTimeAcu = PreTime * (double)FirstTrigFlg;
  587         if (PreDis <= c_PT_long) {
  588           if (fabs(d_PV_tar - b_PV_me) > 1.0E-5) {
  589             Kur = (d_PV_tar - b_PV_me) / c_PT_long;
  590           } else {
  591             Kur = 0.0;
  592           }
  593   
  594           Arc = b_PV_me * PreDis + 0.5 * Kur * (PreDis * PreDis);
  595           V_tarPos = b_PV_me + PreDis * Kur;
  596         } else {
  597           Arc = (b_PV_me + d_PV_tar) / 2.0 * c_PT_long + (PreDis - c_PT_long) *
  598             d_PV_tar;
  599           V_tarPos = d_PV_tar;
  600         }
  601   
  602         if (Arc <= PArclengthTol[0]) {
  603           c_ElementaryPosCal(Arc, PDeltaTemp[0], PArclengthTol[0], Ptheta_qitemp,
  604                              b_Pangle_in, b_Px_in, b_Py_in, Px_qitemp, Py_qitemp,
  605                              xpositionPre, ypositionPre, Theta);
  606           if (Arc <= PArclengthTol[0] / 2.0) {
  607             Kur = Arc * PDeltaTemp[0];
  608           } else {
  609             Kur = PDeltaTemp[0] * Arc - PArclengthTol[0] / 2.0 * PDeltaTemp[0];
  610           }
  611         } else if (Arc <= PArclengthTol[0] + PArclengthTol[1]) {
  612           Arc -= PArclengthTol[0];
  613           c_ElementaryPosCal(Arc, PDeltaTemp[1], PArclengthTol[1], b_Pangle_out,
  614                              Ptheta_qitemp, Px_qitemp, Py_qitemp, b_Px_out,
  615                              b_Py_out, xpositionPre, ypositionPre, Theta);
  616           if (Arc <= PArclengthTol[1] / 2.0) {
  617             Kur = Arc * PDeltaTemp[1];
  618           } else {
  619             Kur = PDeltaTemp[1] * Arc - PArclengthTol[1] / 2.0 * PDeltaTemp[1];
  620           }
  621         } else {
  622           Arc = (Arc - PArclengthTol[0]) - PArclengthTol[1];
  623   
  624           /*              if Time > PT_long */
  625           *xpositionPre = b_Px_out + Arc * cos(b_Pangle_out);
  626           *ypositionPre = b_Py_out + Arc * sin(b_Pangle_out);
  627   
  628           /*              else                 */
  629           /*              end */
  630           /*  Straight line preview point calculate; */
  631           *Theta = 0.0;
  632           Kur = 0.0;
  633         }
  634   
  635         PreDis = (*xpositionPre - Info_Me[0]) * cos(Info_Me[5]) + (*ypositionPre -
  636           Info_Me[1]) * sin(Info_Me[5]);
  637       }
  638   
  639       *Vx = V_tarPos * cos(*Theta);
  640       *Vy = V_tarPos * sin(*Theta);
  641       PreDis = V_tarPos * V_tarPos * Kur;
  642       *Ax = PreDis * cos(*Theta);
  643       *Ay = PreDis * sin(*Theta);
  644   
  645       /* To cal next point */
  646       if (b_PTime + 0.01 <= c_PT_long) {
  647         if (fabs(d_PV_tar - b_PV_me) > 1.0E-5) {
  648           Kur = (d_PV_tar - b_PV_me) / c_PT_long;
  649         } else {
  650           Kur = 0.0;
  651         }
  652   
  653         Arc = b_PV_me * (b_PTime + 0.01) + 0.5 * Kur * ((b_PTime + 0.01) *
  654           (b_PTime + 0.01));
  655       } else {
  656         Arc = (b_PV_me + d_PV_tar) / 2.0 * c_PT_long + ((b_PTime + 0.01) -
  657           c_PT_long) * d_PV_tar;
  658       }
  659   
  660       if (Arc <= PArclengthTol[0]) {
  661         c_ElementaryPosCal(Arc, PDeltaTemp[0], PArclengthTol[0], Ptheta_qitemp,
  662                            b_Pangle_in, b_Px_in, b_Py_in, Px_qitemp, Py_qitemp,
  663                            &Kur, &V_tarPos, &PreDis);
  664   
  665         /*              Pxposition = xpositionPre;Pyposition = ypositionPre; */
  666       } else if (Arc <= PArclengthTol[0] + PArclengthTol[1]) {
  667         Arc -= PArclengthTol[0];
  668         c_ElementaryPosCal(Arc, PDeltaTemp[1], PArclengthTol[1], b_Pangle_out,
  669                            Ptheta_qitemp, Px_qitemp, Py_qitemp, b_Px_out, b_Py_out,
  670                            &Kur, &V_tarPos, &PreDis);
  671   
  672         /*              Pxposition = xpositionPre;Pyposition = ypositionPre; */
  673       } else {
  674         Arc = (Arc - PArclengthTol[0]) - PArclengthTol[1];
  675   
  676         /*              if Time > PT_long */
  677         Kur = b_Px_out + Arc * cos(b_Pangle_out);
  678         V_tarPos = b_Py_out + Arc * sin(b_Pangle_out);
  679   
  680         /*              else                 */
  681         /*              end */
  682         /*  Straight line preview point calculate; */
  683         /*                  if PathPlanRes < 1e-1  */
  684         /*                      if LCTransEn < 0.5 */
  685         /*                          PointCalEn = 0;PrePointCalEn = 0; */
  686         /*                      else */
  687         /*                          PInLaneLCTrans = 1; */
  688         /*                          PathPlanSucFlg = 1; */
  689         /*                      end */
  690         /*                  end */
  691         if (LCTransEn < 0.5) {
  692           if (PathPlanRes < 0.1) {
  693             b_PointCalEn = 0.0;
  694             b_PrePointCalEn = 0.0;
  695           }
  696         } else {
  697           PInLaneLCTrans = 1.0;
  698           *PathPlanSucFlg = 1.0;
  699         }
  700   
  701         *PathTrjFinishFlg = 1.0;
  702       }
  703   
  704       *xpositionNex = PxpositionNex;
  705       *ypositionNex = PypositionNex;
  706       PxpositionNex = Kur;
  707       PypositionNex = V_tarPos;
  708       if (!(PToLeftOrRight > 0.1)) {
  709         *ypositionPre = -*ypositionPre;
  710         *Ay = -*Ay;
  711         *Vy = -*Vy;
  712         *Theta = -*Theta;
  713         *ypositionNex = -*ypositionNex;
  714       }
  715   
  716       *LCInTransState = PInLaneLCTrans;
  717     }
  718   }
