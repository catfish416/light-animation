  806             exitg1 = true;
  807           } else {
  808             ix++;
  809           }
  810         }
  811       }
  812   
  813       if (ixstart < 3) {
  814         while (ixstart + 1 < 4) {
  815           if (varargin_1[ixstart] < alpha) {

  817           }

  820         }

  826       a_me = 0.0;

  837         alpha = a + 0.38196602 * (*xoptres - a);
  838         beta = a + 0.61803398 * (*xoptres - a);
  839         Pfunc(alpha, mtmp, &f_alpha, &a_me);
  840         Pfunc(beta, mtmp, &f_beta, &a_me);

  842           (*IterGoldenStep)++;
  843   
  844           /*      a1=a;  b1=b; */
  845           if (alpha > beta) {
  846             /*          a=a1;b=b1; */
  847             alpha = a + 0.38196602 * (*xoptres - a);

  874         /*       */
  875         Pfunc(a, mtmp, &a_me, &alpha);

  880         }
  881   
  882         /*      if Pfunc(a,LB,UB)<Pfunc(b,LB,UB) */

  884         /*      else */

  886         /*      end */
  887         if (f_alpha >= -0.01) {
