 49function [xposition,yposition,xpositionNex,ypositionNex,Ax,Ay,Vx,Vy,Theta,Time2Cal]=CalXpos2TarPosStop(Info_Me,TarPosInfo,aTrans,PreViewTime,StopReqFlg,V_tar)   
 50%     aTrans = 3;
 51    Time2Nex = 0.01;
 52    TarPos = sqrt((Info_Me(1) - TarPosInfo(1))^2 + ((Info_Me(2) - TarPosInfo(2))^2))+0.0001;
 53    if Info_Me(1)<=TarPosInfo(1)
 54            
 55    else    
 56        %Stop Right Now        
 57        TarPos = 0.1;
 58    end
 59%     if (Info_Me(3)^2/2/TarPos) < (aTrans - 1)
 60%         Time2Acc = sqrt((Info_Me(3)^2 + 2*aTrans*TarPos)/2 - Info_Me(3))/aTrans;
 61% %         Time2Cal = min([Time2Acc,PreViewTime]);
 62%         Time2Cal = PreViewTime;
 63%         if Time2Cal > Time2Acc
 64%             Vx = Info_Me(3) + aTrans * Time2Acc - aTrans * (Time2Cal - Time2Acc);
 65%             Arclen = Info_Me(3) * Time2Acc + aTrans * Time2Acc^2 / 2;
 66%             Arclen = Arclen + (Info_Me(3) + aTrans * Time2Acc) * (Time2Cal - Time2Acc) - aTrans * (Time2Cal - Time2Acc)^2 / 2;
 67%             Ax = -aTrans;
 68%         else
 69%             Vx = Info_Me(3) + aTrans * Time2Cal;
 70%             Arclen = Info_Me(3) * Time2Cal + aTrans * Time2Cal^2 / 2;
 71% %                     Arclen = Info_Me(3) * Time2Cal + aTrans * Time2Cal^2 / 2;   
 72%             Ax = aTrans;
 73%         end        
 74%         if Time2Nex > Time2Acc
 75%             ArclenNex = Info_Me(3) * Time2Acc + aTrans * Time2Acc^2 / 2;
 76%             ArclenNex = ArclenNex + (Info_Me(3) + aTrans * Time2Acc) * (Time2Nex - Time2Acc) - aTrans * (Time2Nex - Time2Acc)^2 / 2;
 77%         else
 78%             ArclenNex = Info_Me(3) * Time2Nex + aTrans * Time2Nex^2 / 2;
 79%         end
 80%         xpositionNex = Info_Me(1) + ArclenNex * cos(Info_Me(6));
 81%         ypositionNex = Info_Me(2) + ArclenNex * sin(Info_Me(6));          
 82%         xposition = Info_Me(1) + Arclen * cos(Info_Me(6));
 83%         yposition = Info_Me(2) + Arclen * sin(Info_Me(6));  
 84%         Vy = 0;
 85%         Ay = 0;
 86%         Theta = Info_Me(6);
 87%     else
 88        % Stop infront of landMark   
 89        if Info_Me(3) < 1e-1
 90            Time2Stop = 0;
 91            Time2Cal = PreViewTime;
 92            ArclenNex = 0;
 93            Arclen = 0;
 94            Ax = -0.5;
 95        else
 96            Ax = -Info_Me(3)^2/2/TarPos;
 97            if Ax < -aTrans
 98                Ax = -aTrans;
 99            elseif Ax > -3e-1
100                Ax = -0.4;
101            else
102            end
103            if abs(Ax)>1e-1
104                Time2Stop = abs(Info_Me(3)/Ax);
105            else
106                Time2Stop = 255;
107            end
108%             Time2Cal = min([Time2Stop,PreViewTime]);
109            Time2Cal = PreViewTime;
110            if Time2Cal > Time2Stop
111                Arclen = Info_Me(3)*Time2Stop + Ax*Time2Stop^2/2;
112            else
113                Arclen = Info_Me(3)*Time2Cal + Ax*Time2Cal^2/2;
114            end
115            if Arclen < 0
116                Arclen = 0;
117            end
118            if Time2Nex > Time2Stop
119                ArclenNex = Info_Me(3) * Time2Stop + Ax * Time2Stop^2 / 2;
120            else
121                ArclenNex = Info_Me(3) * Time2Nex + Ax * Time2Nex^2 / 2;
122            end
123        end
124        Vx = Info_Me(3) + Ax*Time2Cal;
125        if Vx < 0
126            Vx = 0;
127        end
128        Vy = 0;
129        Ay = 0;
130        Theta = Info_Me(6);        
131        xposition = Info_Me(1) + Arclen * cos(Info_Me(6));
132        yposition = Info_Me(2) + Arclen * sin(Info_Me(6));
133        xpositionNex = Info_Me(1) + ArclenNex * cos(Info_Me(6));
134        ypositionNex = Info_Me(2) + ArclenNex * sin(Info_Me(6));          
135%     end
136
137    
138    if StopReqFlg<0.5
139        %Not stop ,just to lower speed
140        Time2Cal = PreViewTime;
141        %V_tar
142        if Info_Me(3)>V_tar
143            Time2V_tar = abs(-Info_Me(3)+V_tar)/aTrans;
144            Ax = sign(-Info_Me(3)+V_tar)*aTrans;
145            if Time2Cal < Time2V_tar
146                Arclen = Info_Me(3)*Time2Cal + Ax*Time2Cal^2/2;
147                Vx = Info_Me(3) + Ax*Time2Cal;
148            else
149                Arclen = Info_Me(3)*Time2V_tar + Ax*Time2V_tar^2/2 + V_tar*(Time2Cal - Time2V_tar);
150                Vx = V_tar;
151            end            
152            Vy = 0;
153            Ay = 0;
154            Theta = Info_Me(6);      
155            if Time2Nex < Time2V_tar
156                ArclenNex = Info_Me(3)*Time2Nex + Ax*Time2Nex^2/2;
157            else
158                ArclenNex = Info_Me(3)*Time2V_tar + Ax*Time2V_tar^2/2 + V_tar*(Time2Nex - Time2V_tar);
159            end
160            xposition = Info_Me(1) + Arclen * cos(Info_Me(6));
161            yposition = Info_Me(2) + Arclen * sin(Info_Me(6));
162            xpositionNex = Info_Me(1) + ArclenNex * cos(Info_Me(6));
163            ypositionNex = Info_Me(2) + ArclenNex * sin(Info_Me(6)); 
164        else
165            Vx = Info_Me(3);
166            Vy = 0;
167            Ay = 0;
168            Ax = 0;
169            Theta = Info_Me(6);      
170            Arclen = Info_Me(3)*Time2Cal;
171            ArclenNex = Info_Me(3)* Time2Nex;
172            xposition = Info_Me(1) + Arclen * cos(Info_Me(6));
173            yposition = Info_Me(2) + Arclen * sin(Info_Me(6));
174            xpositionNex = Info_Me(1) + ArclenNex * cos(Info_Me(6));
175            ypositionNex = Info_Me(2) + ArclenNex * sin(Info_Me(6)); 
176        end
177           
178    end
179end

181function [xposition,yposition,xpositionNex,ypositionNex,Ax,Ay,Vx,Vy,Theta,Time2Cal]=CalXpos2TarPosStart(Info_Me,TarPosInfo,aTrans,PreViewTime,angleout,V_tar)   
182%     aTrans = 3;
183    TarPos = sqrt((Info_Me(1) - TarPosInfo(1))^2 + ((Info_Me(2) - TarPosInfo(2))^2))+0.0001;
184    Time2Nex = 0.01;
185%     if abs(abs(Info_Me(3)) - V_tar)<1e-1
186    if (Info_Me(3) - V_tar)>= 0
187        A_me = 0;Time2Acc = 0;
188    else
189        A_me = (V_tar^2 - Info_Me(3)^2)/2/TarPos;
190        if A_me > aTrans
191            A_me = aTrans;
192        end
193%         Time2Acc = 2*TarPos/(Info_Me(3) + V_tar);
194        Time2Acc = abs(Info_Me(3) - V_tar)/(A_me+0.001);
195    end    
196    
197%     Time2Cal = min([Time2Acc,PreViewTime]);
198    Time2Cal = PreViewTime;
199    if Time2Cal > Time2Acc
200        Vx = V_tar;
201        Arclen = Info_Me(3) * Time2Acc + A_me * Time2Acc^2 / 2 + V_tar * (Time2Cal - Time2Acc);        
202    else
203        Vx = Info_Me(3) + A_me * Time2Cal;
204        Arclen = Info_Me(3) * Time2Cal + A_me * Time2Cal^2 / 2;
205    end
206    if Time2Nex > Time2Acc
207        ArclenNex = Info_Me(3) * Time2Acc + A_me * Time2Acc^2 / 2 + V_tar * (Time2Nex - Time2Acc);        
208    else
209        ArclenNex = Info_Me(3) * Time2Nex + A_me * Time2Nex^2 / 2;  
210    end 
211    
212%     
213%             if Time2Cal > Time2Acc
214%             Vx = Info_Me(3) + aTrans * Time2Acc - aTrans * (Time2Cal - Time2Acc);
215%             Arclen = Info_Me(3) * Time2Acc + aTrans * Time2Acc^2 / 2;
216%             Arclen = Arclen + (Info_Me(3) + aTrans * Time2Acc) * (Time2Cal - Time2Acc) - aTrans * (Time2Cal - Time2Acc)^2 / 2;
217%             Ax = -aTrans;
218%         else

220%             Arclen = Info_Me(3) * Time2Cal + aTrans * Time2Cal^2 / 2;
221% %                     Arclen = Info_Me(3) * Time2Cal + aTrans * Time2Cal^2 / 2;   
222%             Ax = aTrans;
223%         end
224    
225    xposition = Info_Me(1) + Arclen * cos(angleout);
226    yposition = Info_Me(2) + Arclen * sin(angleout);  
227    xpositionNex = Info_Me(1) + ArclenNex * cos(angleout);
228    ypositionNex = Info_Me(2) + ArclenNex * sin(angleout);
