1function [Info_LdPGCd,Info_FdPGCd,Info_MePGCd,x_inPGCd,y_inPGCd,angle_inPGCd,x_outPGCd,y_outPGCd,angle_outPGCd,TrigFlgOut,CurOpPGCd,PGCdInfo,W_inPGCd,W_outPGCd] = CordTransferInterSection(Info_LdGCd,Info_FdGCd,Info_MeGCd,CurOpGCd,x_inGCd,y_inGCd,angle_inGCd,x_outGCd,y_outGCd,angle_outGCd,W_inGCd,W_outGCd,TargetPosInfo,TrigFlgIn)         
 2%#codegen
 3%RoadInfo(1,1) = Info_Me(1);RoadInfo(1,2) = Info_Me(2);RoadInfo(1,3) = Info_Me(6);RoadInfo(1,4) = EgoLaneInfo(2);
 4    persistent CurOpTemp PGCXorig PGCYorig PGCThetaorig
 5    if isempty(CurOpTemp)
 6        CurOpTemp = 0;
 7    end
 8    if isempty(PGCXorig)
 9        PGCXorig = 0;
10    end
11    if isempty(PGCYorig)
12        PGCYorig = 0;
13    end
14    if isempty(PGCThetaorig)
15        PGCThetaorig = 0;
16    end
17    Info_MePGCd = [0 0 0 0 0 0];Info_FdPGCd = [0 0 0 0 0 0];Info_LdPGCd = [0 0 0 0 0 0];PGCdInfo = [0 0 0];CurOpPGCd = CurOpGCd;TrigFlgOut = TrigFlgIn;
18    if abs(CurOpTemp - CurOpGCd) >1e-1
19        CurOpTemp = CurOpGCd;
20        if abs(CurOpGCd - 2) < 1e-1 || abs(CurOpGCd - 1) < 1e-1 
21            PGCXorig = Info_MeGCd(1,1);
22            PGCYorig = Info_MeGCd(1,2);
23            PGCThetaorig = Info_MeGCd(1,6);
24        end
25    else
26    end
27    Info_MePGCd = Info_MeGCd;Info_FdPGCd = Info_FdGCd;Info_LdPGCd = Info_LdGCd;
28    x_inPGCd = x_inGCd;y_inPGCd = y_inGCd;angle_inPGCd = angle_inGCd;
29    x_outPGCd = x_outGCd;y_outPGCd = y_outGCd;angle_outPGCd = angle_outGCd;
30    W_inPGCd = W_inGCd;W_outPGCd = W_outGCd;    
31    % Transform only when there is an operation
32    if abs(CurOpGCd - 2) < 1e-1 || abs(CurOpGCd - 1) < 1e-1 
33        Info_MePGCd(1) = (Info_MeGCd(1) - PGCXorig)*cos(PGCThetaorig) + (Info_MeGCd(2) - PGCYorig)*sin(PGCThetaorig); 
34        Info_MePGCd(2) = (Info_MeGCd(2) - PGCYorig)*cos(PGCThetaorig) - (Info_MeGCd(1) - PGCXorig)*sin(PGCThetaorig); 
35        Info_FdPGCd(1) = (Info_FdGCd(1) - PGCXorig)*cos(PGCThetaorig) + (Info_FdGCd(2) - PGCYorig)*sin(PGCThetaorig); 
36        Info_FdPGCd(2) = (Info_FdGCd(2) - PGCYorig)*cos(PGCThetaorig) - (Info_FdGCd(1) - PGCXorig)*sin(PGCThetaorig); 
37        Info_LdPGCd(1) = (Info_LdGCd(1) - PGCXorig)*cos(PGCThetaorig) + (Info_LdGCd(2) - PGCYorig)*sin(PGCThetaorig); 
38        Info_LdPGCd(2) = (Info_LdGCd(2) - PGCYorig)*cos(PGCThetaorig) - (Info_LdGCd(1) - PGCXorig)*sin(PGCThetaorig); 
39        x_inPGCd = 0; 
40        y_inPGCd = 0;    
41        x_outPGCd = (TargetPosInfo(1) - PGCXorig)*cos(PGCThetaorig) + (TargetPosInfo(2) - PGCYorig)*sin(PGCThetaorig); 
42        y_outPGCd = (TargetPosInfo(2) - PGCYorig)*cos(PGCThetaorig) - (TargetPosInfo(1) - PGCXorig)*sin(PGCThetaorig);         
43        W_inPGCd = W_inGCd;
44        W_outPGCd = W_outGCd;
45        angle_outPGCd = angle_outGCd - PGCThetaorig;
46        angle_inPGCd  = angle_inGCd - PGCThetaorig;
47        %Angle Rot
48        Info_MePGCd(6) = Info_MeGCd(6) - PGCThetaorig;
49        Info_FdPGCd(6) = Info_FdGCd(6) - PGCThetaorig;
50        Info_LdPGCd(6) = Info_LdGCd(6) - PGCThetaorig;      
51        PGCdInfo = [PGCXorig,PGCYorig,PGCThetaorig];         
52    else
53        PGCXorig = 0;
54        PGCYorig = 0;
55        PGCThetaorig = 0;        
56    end 
57end
