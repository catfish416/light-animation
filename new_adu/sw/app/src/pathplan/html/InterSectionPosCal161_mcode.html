248function [dk,dkLamda] = qpfunc( H,c,A,b,x1,y1,w1 )
249%UNTITLED2 Summary of this function goes here
250%   Detailed explanation goes here
251Delta = 0.1;
252P = 0.9;
253E = ones(length(y1),1);
254m = length(b);
255k = 0;
256    while(1)
257        k = k + 1
258        if k == 245
259            k = k;
260        end
261        Y = diag(y1);
262        W = diag(w1);
263        Rho = b - A*x1 + w1;
264        Sigma = c + H*x1 -A'*y1;
265        Gamma = y1'*w1;
266        Mu = Delta * Gamma / m;
267        newMatricL = [-H A';A inv(Y)*W];
268        newMatricR = [c + H*x1 - A'*y1;b-A*x1 + Mu*inv(Y)*E];
269        dxy = inv(newMatricL)*newMatricR;

272        dw = inv(Y)*(Mu*E - Y*W*E - W*dy);

448function [xposition,yposition,Theta] = ElementaryPosCal( Arc,Delta,Arclength,angle_out,angle_in,x_in,y_in,x_out,y_out )
449%UNTITLED Summary of this function goes here
450%   Detailed explanation goes here   
451    Beta = (angle_out - angle_in)/2;
452    theta_rot = angle_in;
453    x_outRot = (x_out - x_in)*cos(theta_rot) + (y_out - y_in)*sin(theta_rot);
454    y_outRot = (y_out - y_in)*cos(theta_rot) - (x_out - x_in)*sin(theta_rot);    
455    if Beta == 0
456        SignBeta = 1;
457    else
458        SignBeta = sign(Beta);
459    end
460    if (Arc < 0)
461        xpositionTemp = Arc * cos(angle_in) + x_in;
462        ypositionTemp = Arc * sin(angle_in) + y_in;
463        Theta = angle_in;
464    elseif (Arc <= Arclength/2)&&(Arc >=0)
465        ulimit = Arc * sqrt(abs(Delta)/2);
466        Theta = Delta * (Arc)^2/2;
467        xpositionTemp = sqrt(2/abs(Delta))*quadcosu2(ulimit);
468        ypositionTemp = sqrt(2/abs(Delta))*quadsinu2(ulimit) * SignBeta;
469    elseif (Arc > Arclength/2)&&(Arc < Arclength)
470        ulimit = (Arclength - Arc) * sqrt(abs(Delta)/2); 
471        %Theta = 2*Beta*SignBeta - Delta * (Arclength - Arc)^2/2; 
472        xpositionMirrTemp = sqrt(2/abs(Delta))*quadcosu2(ulimit);
473        ypositionMirrTemp = sqrt(2/abs(Delta))*quadsinu2(ulimit) * SignBeta;
474        %mirror line Ax+By+C=0
475        x_mirr = (x_outRot)/2;
476        y_mirr = (y_outRot)/2;
477        angle_mirr = pi/2 + atan((y_outRot)/(x_outRot));
478        A = tan(angle_mirr);
479        B = -1;
480        C = y_mirr - A * x_mirr;      
481        %two points are symmetrical to the Ax+By+C we just defined
