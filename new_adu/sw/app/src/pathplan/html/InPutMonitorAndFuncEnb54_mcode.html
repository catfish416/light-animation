1function [MonitorSts,LCEnbRes,TunEnbRes,InputStatusFailCnt] = InPutMonitorAndFuncEnb(LCManuEnb,TunManuEnb,TrigStart,DebncCntInit,Navi_InfoTurn,Line_ShapeInfo,Lane_NumInfo,Veh_State,BypassOffsetMon,TTTol,OffsetAcy,LCAngInitDev,LCCurDev,DisEngFlg,TunTolR)
  2%#codegen
  3    persistent TunEnbDebncCnt LCEnbDebncCnt OffsetMonitorCnt SpdAvg InitOffset OffsetMonitorRes Veh_StatusCnt PInputStatusFailCnt
  4    if isempty(TunEnbDebncCnt)
  5        TunEnbDebncCnt = DebncCntInit;
  6        LCEnbDebncCnt = DebncCntInit;
  7        OffsetMonitorCnt = 100;
  8        OffsetMonitorRes = 1;
  9        SpdAvg = 0;
 10        InitOffset = 0;
 11        Veh_StatusCnt = 20;
 12        PInputStatusFailCnt = zeros(3,1);
 13    end    
 14    TunEnbRes = 0; LCEnbRes = 0; MonitorSts = 0; Veh_Status = 0; InputStatusFailCnt = zeros(3,1);
 15
 16    if TrigStart > 0.5 
 17        % 00 - module ill 
 18            % 11 - module LC ill 
 19            % 12 - module Tun ill 
 20        % 10 - module healthy
 21        if BypassOffsetMon > 0.5
 22            OffsetMonitorRes = 1;
 23        else
 24            if OffsetMonitorCnt > 99.5
 25                InitOffset = Veh_State(4);
 26                SpdAvg = 1/100*Veh_State(8);            
 27            elseif OffsetMonitorCnt < 0.5        
 28                OffsetMonitorCnt = 100;
 29                AccuOffset = Veh_State(4) - InitOffset;
 30                Bias = abs(AccuOffset - SpdAvg*100*10/1000);
 31                if Bias < AccuOffset * 0.5
 32                    OffsetMonitorRes = 1;
 33                else
 34                    % Offset not right
 35                    OffsetMonitorRes = 0;
 36                end
 37            else
 38                OffsetMonitorCnt = OffsetMonitorCnt - 1;
 39                SpdAvg = SpdAvg + 1/100*Veh_State(8); 
 40            end  
 41        end
 42
 43        if abs(Veh_State(1)) > 1e5 || abs(Veh_State(2)) > 1e5  
 44            Veh_StatusLo = 0;
 45        else
 46            Veh_StatusLo = 1;
 47        end
 48        if Veh_StatusLo < 0.5
 49            Veh_StatusCnt = 20;
 50        else            
 51            if Veh_StatusCnt < 0.5
 52                Veh_Status = 1;
 53            else
 54                Veh_StatusCnt = Veh_StatusCnt - 1; 
 55            end
 56        end 
 57            % Tun Func Status
 58        TunEnbResTemp = 1; 
 59        if TunManuEnb > 0.5 && Veh_Status > 0.5
 60            if Navi_InfoTurn(1,2) < 0.5 || Navi_InfoTurn(2,2) < 0.5 
 61                % TunTrig will be forbidden
 62                TunEnbResTemp = 0;  
 63            else
 64                TunOffset = Navi_InfoTurn(2,1) - Navi_InfoTurn(1,1);
 65                if TunOffset <= TunTolR || TunOffset > 100 || abs(sin(Navi_InfoTurn(2,7) - Navi_InfoTurn(1,7)))<0.5
 66                    %Bypass small turn 
 67                    TunEnbResTemp = 0;
 68                else
 69                    if Navi_InfoTurn(3,2) < 0.5 || Navi_InfoTurn(4,2) < 0.5
 70                    else
 71                        % Next turn is coming
 72                        TunTunOffset = Navi_InfoTurn(3,1) - Navi_InfoTurn(2,1);
 73                        if TunTunOffset < TTTol 
 74                            TunEnbResTemp = 0;
 75                        end
 76                    end                            
 77                end
 78                DisToTurnIn = sqrt((Navi_InfoTurn(1,3) - Veh_State(1))^2 + (Navi_InfoTurn(1,4) - Veh_State(2))^2);
 79                OffToTurnIn = abs(Navi_InfoTurn(1,1) - Veh_State(4));
 80                if (DisToTurnIn > (OffToTurnIn + OffsetAcy))||(DisToTurnIn < (OffToTurnIn - OffsetAcy))
 81                    TunEnbResTemp = 0;
 82                end     
 83                if DisEngFlg > 0.5
 84                    TunEnbResTemp = 0;
 85                end
 86            end
 87        else
 88            TunEnbResTemp = 0;
 89        end
 90        % Debounce
 91        if TunEnbResTemp < 0.5        
 92            TunEnbDebncCnt = DebncCntInit;
 93        else
 94            if TunEnbDebncCnt < 0.5
 95                TunEnbRes = 1;
 96            else
 97                TunEnbDebncCnt = TunEnbDebncCnt - 1;
 98            end        
 99        end    
100        % LC Func Status
101        LCEnbResTemp = 1; 
102        if LCManuEnb > 0.5 && Veh_Status > 0.5
103            if Lane_NumInfo(1) < 0.5 || Lane_NumInfo(3) < 0.5|| Lane_NumInfo(2) < 0.5
104                LCEnbResTemp = 0;
105            else
106                for i = 1:Lane_NumInfo(1)                                              
107                    if i > 10 || i < 0.5
108                        LCEnbResTemp = 0;
109                        break;
110                    else
111                        if Line_ShapeInfo(i,5)<10
112                            LCEnbResTemp = 0;
113                            break;
114                        elseif Line_ShapeInfo(i,6)<2
115                            LCEnbResTemp = 0;
116                            break;
117                        elseif abs(Lane_NumInfo(2) - i)<1e-1 && (abs(Line_ShapeInfo(i,1)) > 10 || abs(Line_ShapeInfo(i,2)) > 10)
118                            LCEnbResTemp = 0;
119                            break;
120                        elseif abs(Line_ShapeInfo(i,2))>LCAngInitDev
121                            LCEnbResTemp = 0;
122                            break;
123                        elseif abs(2*Line_ShapeInfo(i,3))>LCCurDev
124                            LCEnbResTemp = 0;
125                            break;
126                        else
127                        end
128                    end
129                end
130                if DisEngFlg > 0.5
131                    LCEnbResTemp = 0;
132                end
133            end
134        else
135            LCEnbResTemp = 0;            
136        end
137        % Debounce
138        if LCEnbResTemp < 0.5        
139            LCEnbDebncCnt = DebncCntInit;
140        else
141            if LCEnbDebncCnt < 0.5
142                LCEnbRes = 1;
143            else
144                LCEnbDebncCnt = LCEnbDebncCnt - 1;
145            end        
146        end
147            % Monitor Status Update
148        MonitorSts = 0; 
149        if  Veh_Status < 0.5
150            MonitorSts = -10; 
151            PInputStatusFailCnt(3) = PInputStatusFailCnt(3) + 1;
152        else               
153            if LCEnbRes < 0.5 && TunEnbRes < 0.5            
154            else            
155                if LCEnbRes > 0.5
156                    MonitorSts = 11;   
157                else
158                    PInputStatusFailCnt(1) = PInputStatusFailCnt(1) + 1;
159                end
160                if TunEnbRes > 0.5
161                    MonitorSts = 12;
162                    if LCEnbRes > 0.5
163                        MonitorSts = 10;
164                    end
165                else
166                    PInputStatusFailCnt(2) = PInputStatusFailCnt(2) + 1;
167                end            
168            end  
169        end
170    else
171        TunEnbDebncCnt = DebncCntInit;
172        LCEnbDebncCnt = DebncCntInit;
173        OffsetMonitorCnt = 100;
174        PInputStatusFailCnt = zeros(3,1);
175        Veh_StatusCnt = 20;
176    end
177    
178    InputStatusFailCnt = PInputStatusFailCnt;
179    
180end
