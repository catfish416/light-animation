194function [Dis,Dir] = DisCal2Lane(LaneAng,x1,y1,xobj,yobj)
195%      A*x + B*y + C
196%d = -----------------
197%      sqrt(A^2 + B^2)
198%
199%Dir Front 1 Rear 0 Left 1 Right 0
200%example 11 FL 10 FR 1 RL 0 RR
201%Xdirection Front and Rear Ydirection Left and Right
202    
203    while(1)
204       if LaneAng < -pi
205           LaneAng = LaneAng + 2*pi;
206       elseif LaneAng > pi
207           LaneAng = LaneAng - 2*pi;
208       else           
209            if LaneAng > pi/2
210                LaneAng = LaneAng - pi;
211            elseif LaneAng < -pi/2
212                LaneAng = LaneAng + pi;
213            else
214            end           
215            break;
216       end
217    end
218%Convert LaneAng to -pi/2 - pi/2    
219
220    if abs(sin(LaneAng))>1.414
221        InvK = cot(LaneAng);
222        A = 1;B = -InvK;C = InvK*y1 - x1;
223    else
224        K = tan(LaneAng);
225        A = K;B = -1;C = y1-K*x1;
226    end
227    
228    Dis = (A*xobj + B*yobj + C)/sqrt(A^2 + B^2);
229    %Left or Right
230    if abs(A)<abs(B)
231        if (B * Dis > 0)
232            Dir = 0;
233        else
234            Dir = 1;
235        end    
236    else
237        if (A * Dis > 0)
238            %Right
239            Dir = 1;
240        else
241            %Left
242            Dir = 0; 
243        end
244    end
245    
246    Dis = abs(Dis);    
247end

249function [Dis,Dir] = DisCal2Arc(kur,xobj,yobj)
250    if abs(kur) < 1e-3
251        % Treat as a straight lane
252        [Dis,Dir] = DisCal2Lane(0,0,0,xobj,yobj);
253    else
254        % Treat as a circle
255        CircleX = 0;CircleY = 1/kur;
256        Dis = sqrt((xobj-CircleX)^2+(yobj-CircleY)^2)-abs(CircleY);
257        if kur*Dis>0
258            %Right
259            Dir = 1;
260        else
261            %Left
262            Dir = 0;
263        end        
264    end
265end
