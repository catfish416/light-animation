235function [Distance,theta] = AngleDisCalin2Pi(Xdis,Ydis,EgoSpd)
236    theta = 0;
237    Xdis = Xdis + EgoSpd;
238    Distance = sqrt((Xdis)^2 + Ydis^2);
239    if abs(Xdis) < 1e-5
240        if Ydis < 0
241            theta = 3*pi/2;
242        else
243            theta = pi/2;
244        end
245    else
246        theta = atan(Ydis/Xdis);
247        if abs(theta)<1e-5
248            if Xdis < 0
249                theta = pi;
250            else
251                theta = 0;
252            end
253        else
254            if Xdis < 0
255                theta = theta + pi;
256            else
257                if theta < 1e-1
258                    theta = theta + 2*pi;
259                end
260            end                
261        end
262    end
263    
264    if Xdis < 0
265        Distance = Distance * -1;              
266    end
267    
268end

345function Ind = FusIndSeletFdLd(ToLeft,TargetPosition,Info_Me,Info_Lo,LateralSafetyTol,FusData,LineAng,CalDis,ObjNum,Type,Veh_State)
346%     ToLeft
347%     1    Left
348%     0    Right
349%     Type   
350%     1    Info_Lo = [65535,65535,65535,1.5,2.5,Veh_State(3)]; 
351%     2    Info_Fo = [-65535,-65535,65535,1.5,2.5,Veh_State(3)];
352%     3    Info_Fd = [-65535,-65535,65535,1.5,2.5,Veh_State(3)];
353%     4    Info_Ld = [65535,65535,65535,1.5,2.5,Veh_State(3)];
354    IndTemp = 128;
355    AbsDis = 255;
356    % Search for ld and fd
357    k = 1;
358    while(1)
359        if (k > 127) || (FusData(k,1)< 0.5)                        
360            break;
361        end
362        ObjIndex = k;
363        Info_Temp = ObjInfoMaker(ObjIndex,FusData,ObjNum,Veh_State,Type);        
364        %  when cal Ld, x must > 0, when cal fd, x must < 0
365        if (abs(Type-3) < 0.5 && FusData(k,2) > 0)||(abs(Type-4) < 0.5 && FusData(k,2) <= 0)            
366        else
367            YoffsetTmp = (Info_Temp(1) - CalDis)*tan(LineAng);
368          
369            if ToLeft > 0.5
370                OuterLim = TargetPosition + (Info_Temp(5) + Info_Me(5))/2 + LateralSafetyTol*1.5;
371                InnerLim = Info_Lo(2) + (Info_Temp(5) + Info_Lo(5))/2 + LateralSafetyTol/2;
372                if ((Info_Temp(2)-YoffsetTmp) < OuterLim )&&((Info_Temp(2)-YoffsetTmp) > InnerLim)             
373                    if abs(Info_Temp(1)) < AbsDis
374                        AbsDis = abs(Info_Temp(1));
375                        IndTemp = k;
376                    end
377                else
378                end
379            else
380                OuterLim = Info_Lo(2) - (Info_Temp(5) + Info_Lo(5))/2 - LateralSafetyTol/2;
381                InnerLim = TargetPosition - (Info_Temp(5) + Info_Me(5))/2 - LateralSafetyTol*1.5;
382                if ((Info_Temp(2)-YoffsetTmp) < OuterLim )&&((Info_Temp(2)-YoffsetTmp) > InnerLim)                
383                    if abs(Info_Temp(1)) < AbsDis
384                        AbsDis = abs(Info_Temp(1));
385                        IndTemp = k;
386                    end     
387                else
388                end
389            end
390        end
391        k = k + 1;
392    end

472        if TargetIndex < Lane_NumInfo(1) + 0.5

596function IdArrayRes = BubbleSortSurObjId(Object_Info,IdArray)
597    for j = 1:length(IdArray(:,1))
598        SortLen = length(IdArray(j,:));
599
600        for i =1:length(IdArray(j,:))
601            if IdArray(j,i)<0.5
602                SortLen = i - 1;
603                break;
604            end                
605        end
606        if SortLen < 0.5
607        else
608            %First Sort
609            for i=1:SortLen
610                SortFlg = 0;
611                for k=0:SortLen
612                    Index = SortLen - k;
613                    if Index > i
614                        if abs(Object_Info(IdArray(j,Index),2)) < abs(Object_Info(IdArray(j,Index-1),2))
615                        %Swap
616                            SortFlg = 1; 
617                            [IdArray(j,Index),IdArray(j,Index-1)] = swapVal(IdArray(j,Index),IdArray(j,Index-1));
618                        end
619                    else
620                        break;
621                    end
622                end
623                if SortFlg > 0.5
624                else
625                    break;
626                end
627            end
628        end
629    end
630    IdArrayRes = IdArray;
631end
632function [Ares,Bres] =swapVal(A,B)
633    Ares = B;
634    Bres = A;

636function DisToLane = LaneFuncCal(Line_ShapeInfo,CalLineNum,CalDis)

638        CalDis = min([Line_ShapeInfo(CalLineNum,5),50]);
639    elseif CalDis < 0
640        CalDis = 0;
641    else        
642    end
643    DisToLane = Line_ShapeInfo(CalLineNum,1)+Line_ShapeInfo(CalLineNum,2)*CalDis+Line_ShapeInfo(CalLineNum,3)*CalDis^2+Line_ShapeInfo(CalLineNum,4)*CalDis^3;   
644    DisToLane = -1*DisToLane;
